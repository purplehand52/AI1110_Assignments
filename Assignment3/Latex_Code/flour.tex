%Problem Declaration---------------------------------------
\begin{problem}[15.11]
  
  %Problem---------------------------------------------------
  Eleven bags of wheat flour, each marked 5 kg, actually contained the following weights of flour (in kg):
  \begin{table}[!ht]
    \centering
    \input{flour_table.tex}
    \caption{Weights of Bags}
    \label{table:Para_table}	
  \end{table}
  Find the probability that any of these bags chosen at random conatins more than 5 kg of flour. 
\end{problem}

%Solution--------------------------------------------------
\solution

%Random Variable-------------------------------------------
We will define random variable $X$ such that $X \in {0, 1}$. We will assign $X = 1$ when the weight of the bag exceeds 5 and $X = 0$ otherwise. The resulting random variables for corresponding weights is given by Table
\eqref{table:RV_table}.
\begin{table}[!ht]
  \centering
  \input{rv_table.tex}
  \caption{Random Variables}
  \label{table:RV_table}	
\end{table}


%Probability-----------------------------------------------
We are required to find the probability when random variable $X = 1$.
Note that the number of bags with $X = 1$ is given by Table
\eqref{table:RV_table}
as
\begin{align}
  n(X = 1) = 7
  \label{eq:FavOutcomes}
\end{align}

The sample space $S$ consists of 11 bags. Hence
\begin{align}
  n(S) = 11
  \label{eq:TotOutcomes}
\end{align}

Hence the probability where $X = 1$ from
\eqref{eq:FavOutcomes}
and
\eqref{eq:TotOutcomes}
is given by
\begin{align}
  P(X = 1) = \frac{n(X = 1)}{n(S)}
  \\
  =\frac{7}{11}
  \\
  =\text{\fbox{0.636}}
\end{align}
